## 1. 通用规范

### 1.1 Git 分支策略

```
main            永久分支，始终可部署到生产，受保护（不允许直接推送）
release/*       发布分支，从 develop 切出，只接受 bugfix
develop         集成分支，所有功能分支合并目标
feature/*       功能分支，命名：feature/MAAS-{issue_id}-{简要描述}
bugfix/*        修复分支，命名：bugfix/MAAS-{issue_id}-{简要描述}
hotfix/*        生产热修，从 main 切出，合并回 main 和 develop

示例：
  feature/MAAS-123-add-streaming-support
  bugfix/MAAS-456-fix-rate-limit-reset
  hotfix/MAAS-789-fix-billing-overflow
```

### 1.2 Commit Message 规范

遵循 [Conventional Commits](https://www.conventionalcommits.org/) 规范：

```
格式：<type>(<scope>): <subject>

type:
  feat     新功能
  fix      Bug 修复
  docs     文档变更
  style    格式调整（不影响功能）
  refactor 重构（非新功能也非 Bug 修复）
  perf     性能优化
  test     测试相关
  chore    构建/工具链/依赖变更
  revert   回滚

scope（可选）：gateway | routing | adapter | billing | auth | finetune | cache | monitor | notification | frontend

示例：
  feat(gateway): add streaming SSE proxy support
  fix(billing): fix duplicate charge on retry requests
  docs(api): update chat completions request schema
  perf(cache): optimize vector search with HNSW index
  test(auth): add unit tests for API key validation
```

### 1.3 Code Review 规范

```
PR 要求：
- 标题遵循 Commit Message 格式
- 描述模板：What（做了什么）/ Why（为什么）/ How（如何测试）
- 不超过 400 行变更（超出需拆分）
- 至少 1 名 Reviewer 批准才可合并
- 核心服务（gateway/billing/auth）需 2 名批准

Review Checklist：
□ 逻辑正确性：是否实现了预期功能？
□ 安全性：是否有注入风险/权限问题/敏感信息泄露？
□ 性能：是否有 N+1 查询/无必要的全量扫描？
□ 错误处理：异常情况是否都被处理？
□ 测试覆盖：是否有对应单元测试/集成测试？
□ 日志：关键操作是否有合适的日志记录？
□ 文档：是否需要更新相关文档？
```

## 2. Go 代码规范

### 2.1 基础规范

遵循 [Effective Go](https://go.dev/doc/effective_go) + [Google Go Style Guide](https://google.github.io/styleguide/go/)，以下为项目特定约定：

```go
// ✅ 包名：小写单词，不使用下划线或混合大小写
package gateway

// ✅ 常量：使用 const 块，驼峰命名
const (
    MaxRetries    = 3
    DefaultTimeout = 30 * time.Second
)

// ✅ 错误定义：在包级别统一定义，以 Err 开头
var (
    ErrInvalidAPIKey    = errors.New("invalid api key")
    ErrTenantSuspended  = errors.New("tenant suspended")
    ErrRateLimitExceeded = errors.New("rate limit exceeded")
)

// ✅ 接口命名：以 -er 结尾（描述行为）
type APIKeyValidator interface {
    ValidateAPIKey(ctx context.Context, rawKey string) (*AuthInfo, error)
}

// ✅ Struct 字段顺序：必填字段在前，可选字段在后，同类型字段相邻
type ChatRequest struct {
    Model    string    // 必填
    Messages []Message // 必填
    MaxTokens *int     // 可选
    Stream   bool      // 可选，默认值字段在最后
}
```

### 2.2 错误处理

```go
// ✅ 正确：包装错误，保留上下文
func (s *RoutingService) SelectEndpoint(ctx context.Context, req *RoutingRequest) (*Endpoint, error) {
    endpoints, err := s.repo.ListAvailableEndpoints(ctx, req.ModelID)
    if err != nil {
        return nil, fmt.Errorf("routing: list endpoints for model %s: %w", req.ModelID, err)
    }
    if len(endpoints) == 0 {
        return nil, fmt.Errorf("routing: no available endpoints for model %s: %w", req.ModelID, ErrNoAvailableEndpoint)
    }
    // ...
}

// ❌ 错误：吞掉错误
endpoints, _ := s.repo.ListAvailableEndpoints(ctx, req.ModelID)

// ❌ 错误：仅打印不返回
if err != nil {
    log.Printf("error: %v", err)
}
```

### 2.3 上下文与超时

```go
// ✅ 所有外部调用必须传递 context，并设置超时

    ctx, cancel := context.WithTimeout(ctx, 30*time.Second)
```

### 2.4 并发安全

```go
// ✅ 使用 sync.RWMutex 保护共享状态
type EndpointCache struct {
    mu        sync.RWMutex
    endpoints map[string]*Endpoint
}

func (c *EndpointCache) Get(id string) (*Endpoint, bool) {
    c.mu.RLock()
    defer c.mu.RUnlock()
    ep, ok := c.endpoints[id]
    return ep, ok
}

// ✅ 优先使用 channel 传递数据，而非共享内存
func (s *StreamProcessor) Process(ctx context.Context) <-chan Chunk {
    out := make(chan Chunk, 10)
    go func() {
        defer close(out)
        // ...
    }()
    return out
}
```

### 2.5 测试规范

```go
// ✅ 单元测试文件命名：xxx_test.go，与被测文件同目录
// ✅ 测试函数命名：Test{FunctionName}_{Scenario}
func TestValidateAPIKey_ValidKey(t *testing.T) { ... }
func TestValidateAPIKey_ExpiredKey(t *testing.T) { ... }
func TestValidateAPIKey_RevokedKey(t *testing.T) { ... }

// ✅ 使用 Table-Driven Tests
func TestCalculateCost(t *testing.T) {
    tests := []struct {
        name         string
        inputTokens  int
        outputTokens int
        pricePerK    float64
        want         float64
    }{
        {"basic", 1000, 500, 0.002, 3.0},
        {"zero_tokens", 0, 0, 0.002, 0.0},
        {"large_input", 100000, 50000, 0.002, 300.0},
    }

    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got := CalculateCost(tt.inputTokens, tt.outputTokens, tt.pricePerK)
            assert.Equal(t, tt.want, got)
        })
    }
}

// 覆盖率要求：
// - 核心业务逻辑（billing/auth/routing）：>= 80%
// - 工具函数、Adapter：>= 70%
// - Handler 层（HTTP/gRPC）：>= 60%
```

## 3. Python 代码规范（finetune-service）

### 3.1 基础规范

```python
# 遵循 PEP 8，以下为项目特定约定

# ✅ 类型注解：所有公共函数必须有类型注解
async def submit_finetune_job(
    tenant_id: str,
    config: FinetuneConfig,
    dataset_path: str,
) -> FinetuneJob:
    ...

# ✅ 文档字符串：使用 Google 风格
def calculate_training_steps(num_samples: int, batch_size: int, epochs: int) -> int:
    """计算训练总步数。

    Args:
        num_samples: 训练样本总数
        batch_size: 批次大小
        epochs: 训练轮次

    Returns:
        总训练步数

    Raises:
        ValueError: 当 batch_size <= 0 时
    """
    if batch_size <= 0:
        raise ValueError(f"batch_size must be positive, got {batch_size}")
    return (num_samples // batch_size) * epochs
```

## 4. TypeScript / React 代码规范

### 4.1 TypeScript 规范

```typescript
// ✅ 优先使用 interface 定义对象类型（可扩展）
interface ChatMessage {
  role: "system" | "user" | "assistant";
  content: string;
  timestamp?: number;
}

// ✅ 使用 type 定义联合类型或工具类型
type APIStatus = "idle" | "loading" | "success" | "error";
type Nullable<T> = T | null;

// ✅ 函数参数超过 2 个时使用对象参数
function createApiKey(options: {
  name: string;
  scopes: string[];
  expiresAt?: Date;
}): Promise<ApiKey> { ... }

// ❌ 禁止使用 any
const data: any = response; // ❌

// ✅ 使用 unknown 并做类型守卫
const data: unknown = response;
if (isApiResponse(data)) {
  console.log(data.model); // 安全访问
```

### 4.2 React 组件规范

```tsx
// ✅ 函数组件 + 类型显式声明
interface ApiKeyCardProps {
  apiKey: ApiKey;
  onRevoke: (id: string) => void;
}

const ApiKeyCard: React.FC<ApiKeyCardProps> = ({ apiKey, onRevoke }) => {
  // ✅ 状态集中管理，props 不超过 5 个
  const [loading, setLoading] = useState(false);

  // ✅ 事件处理函数以 handle 开头
  const handleRevoke = useCallback(async () => {
    setLoading(true);
    try {
      await onRevoke(apiKey.id);
    } finally {
      setLoading(false);
    }
  }, [apiKey.id, onRevoke]);

  return (
    <Card>
      <Text code>{apiKey.maskedKey}</Text>
      <Button danger loading={loading} onClick={handleRevoke}>
        吊销
      </Button>
    </Card>
  );
};

export default ApiKeyCard;
```

## 5. SQL 规范

```sql
-- ✅ 所有查询必须携带 tenant_id 过滤（防越权）
SELECT id, name, status
FROM api_keys
WHERE tenant_id = $1        -- 必须的租户隔离条件
  AND deleted_at IS NULL
ORDER BY created_at DESC
LIMIT 20 OFFSET $2;

-- ✅ 新增表必须包含标准审计字段
CREATE TABLE example_table (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id   UUID NOT NULL REFERENCES tenants(id),
    -- ... 业务字段 ...
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
    deleted_at  TIMESTAMPTZ,  -- 软删除
    created_by  UUID REFERENCES users(id)
);

-- ✅ 使用参数化查询，禁止字符串拼接
-- ❌ 禁止：db.Query("SELECT * FROM users WHERE name = '" + name + "'")
-- ✅ 正确：db.Query("SELECT * FROM users WHERE name = $1", name)
```

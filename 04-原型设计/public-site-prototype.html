from openai import OpenAI

# 替换为你的 API Key
client = OpenAI(
    api_key="sk-maas-xxxxxxxx",
    base_url="https://api.maas.io/v1"
)

    model="gpt-4o",  # 或任意模型
    messages=[{
        "role": "user",
        "content": "你好！"
    }]
)
print(response.choices[0].message.content)
